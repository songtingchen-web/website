2. Core Financial Datasets (2022–2025)

View Code (Python) 
#data
import pandas as pd

# --- 1. Dataset Initialization ---
# Defining historical and projected financial data for analysis
financial_data = {
    'Year': [2022, 2023, 2024, 2025],
    'Revenue': [185000, 210000, 235000, 268000],
    'COGS': [95000, 108000, 119000, 132000],
    'Op_Expenses': [61500, 67800, 74500, 86200],
    'Net_Profit': [28500, 34200, 41500, 49800],
    'Current_Assets': [46000, 52000, 59000, 67000],
    'Current_Liabilities': [22000, 24000, 26000, 29000]
}

df = pd.DataFrame(financial_data)

# --- 2. Financial Calculation Logic ---
# Deriving key profitability and liquidity metrics 
df['Gross_Profit'] = df['Revenue'] - df['COGS']
df['GPM'] = (df['Gross_Profit'] / df['Revenue']) * 100
df['NPM'] = (df['Net_Profit'] / df['Revenue']) * 100
df['Current_Ratio'] = df['Current_Assets'] / df['Current_Liabilities']

# --- 3. Strategic KPI Summary Report ---
# Formatting the output table to align with professional accounting standards
styled_df = df[['Year', 'GPM', 'NPM', 'Current_Ratio']].copy()
styled_df.columns = ['Fiscal Year', 'Gross Margin (%)', 'Net Margin (%)', 'Liquidity (Ratio)']

# Print formatted summary for the ePortfolio display
print("\n" + "=" * 70)
print(f"{'STRATEGIC KPI SUMMARY':^70}")
print("=" * 70)
print(styled_df.to_string(index=False, float_format=lambda x: f'{x:.2f}'))
print("=" * 70)

3. Profitability Engine: Dual-Axis Trend Analysis

View Code (Python) 
# Reduce figsize to (7, 4.5) for a more compact output
fig, ax1 = plt.subplots(figsize=(7, 4.5))

# Plot Revenue bars (using defined NAVY color)
bars = ax1.bar(df['Year'], df['Revenue'], color=NAVY, alpha=0.85, label='Total Revenue', width=0.4)
ax1.set_ylabel('Revenue (EUR)', fontweight='bold', color=NAVY, fontsize=9)
ax1.tick_params(axis='y', labelcolor=NAVY, labelsize=8)

# Create secondary y-axis for profit margin
ax2 = ax1.twinx()
ax2.plot(df['Year'], df['NPM'], color=GOLD, marker='D', linewidth=2.5, markersize=6, label='Net Profit Margin (%)')
ax2.set_ylabel('Net Margin (%)', fontweight='bold', color=GOLD, fontsize=9)
ax2.tick_params(axis='y', labelcolor=GOLD, labelsize=8)

# Aesthetic optimizations
plt.title('Performance Scaling: Revenue vs. Profitability', fontsize=11, fontweight='bold', pad=20)
ax1.set_xticks(df['Year'])
ax1.tick_params(axis='x', labelsize=8)
ax1.grid(axis='y', linestyle='--', alpha=0.2)

# Merge legends and adjust font size
lines, labels = ax1.get_legend_handles_labels()
lines2, labels2 = ax2.get_legend_handles_labels()
ax2.legend(lines + lines2, labels + labels2, loc='upper left', frameon=False, fontsize=8)

plt.tight_layout()
plt.show()
  • Strategic Insight: Net margins expanded to 18.6% alongside revenue growth, confirming strong economies of scale and optimized operational leverage.

4. Cost Structure Breakdown (Waterfall Analysis)

View Code (Python) 
latest = df.iloc[-1]

# Construct Plotly waterfall object with optimized dimensions
fig = go.Figure(go.Waterfall(
    name = "2025 Financials", orientation = "v",
    measure = ["relative", "relative", "relative", "total"],
    x = ["Revenue", "COGS", "Op Exp", "Net Profit"], # Abbreviated labels
    textposition = "outside",
    text = [f"+{latest['Revenue']}", f"-{latest['COGS']}", f"-{latest['Op_Expenses']}", f"{latest['Net_Profit']}"],
    y = [latest['Revenue'], -latest['COGS'], -latest['Op_Expenses'], latest['Net_Profit']],
    connector = {"line":{"color":"rgb(63, 63, 63)"}},
    increasing = {"marker":{"color": NAVY}},
    decreasing = {"marker":{"color": GOLD}},
    totals = {"marker":{"color": DARK_GREY}},
    textfont=dict(size=10) # Reduce text size on bars
))

fig.update_layout(
    title = dict(text="Profit Erosion Mapping (2025 FY)", font=dict(size=13)),
    showlegend = False,
    plot_bgcolor = 'white',
    width=550, height=380, # Compressed container size
    margin=dict(t=60, b=40, l=40, r=40),
    xaxis=dict(tickfont=dict(size=10)),
    yaxis=dict(tickfont=dict(size=10))
)

fig.show()
